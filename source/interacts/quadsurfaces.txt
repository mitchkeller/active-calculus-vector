from sage.plot.plot3d.plot3d import axes
var('x,y,z,t,s')
@interact
def quadsurfaces(sb=selector(['Ellipsoid','Hyperboloid of One Sheet','Hyperboloid of Two Sheets','Elliptic Paraboloid','Saddle/Hyperbolic Paraboloid','Cone'],label="Surface Options:")):
    if sb=='Ellipsoid':
        scale=2
        a=1;b=1.5;c=1.25;
        param(s,t)=(a*sin(s)*cos(t),b*sin(s)*sin(t),c*cos(s));s1=0;s2=pi;t1=0;t2=2*pi;
        xtparam(xv,t)=(xv,b*sqrt(1-xv^2/a^2)*cos(t),c*sqrt(1-xv^2/a^2)*sin(t))
        ytparam(yv,t)=(a*sqrt(1-yv^2/b^2)*cos(t),yv,c*sqrt(1-yv^2/b^2)*sin(t))
        ztparam(zv,t)=(a*sqrt(1-zv^2/c^2)*cos(t),b*sqrt(1-zv^2/c^2)*sin(t),zv)
        n=6; xt1=0;xt2=2*pi; yt1=0;yt2=2*pi; zt1=0;zt2=2*pi;
        xvals=[-a+i/n*2*a for i in range(1,n)]; yvals=[-b+i/n*2*b for i in range(1,n)] ; zvals=[-b+i/n*2*b for i in range(1,n)]
        plot30=axes(scale,1.5,color="gray",frame=False,aspect_ratio=1)+text3d("x",(scale+0.5,0,0),fontsize=30)+text3d("y",(0,scale+0.5,0),fontsize=30)+text3d("z",(0,0,scale+0.5),fontsize=30)
        plot30+=parametric_plot3d(param,(s,s1,s2),(t,t1,t2),color="gray",opacity=0.2)
        for xtrace in xvals:
            plot30+=parametric_plot3d(xtparam(xtrace,t),(t,xt1,xt2),color="blue",thickness=1)
        for ytrace in xvals:
            plot30+=parametric_plot3d(ytparam(ytrace,t),(t,yt1,yt2),color="green",thickness=1)
        for ztrace in zvals:
            plot30+=parametric_plot3d(ztparam(ztrace,t),(t,zt1,zt2),color="red",thickness=1)    
    if sb=='Hyperboloid of One Sheet':
        scale=3
        a=1;b=1.25;c=1;
        param(s,t)=(a*sqrt(s^2+1)*cos(t),b*sqrt(s^2+1)*sin(t),c*s);s1=-3;s2=3;t1=0;t2=2*pi;
        plot30=axes(scale,1.5,color="gray",frame=False,aspect_ratio=1)+text3d("x",(scale+0.5,0,0),fontsize=30)+text3d("y",(0,scale+0.5,0),fontsize=30)+text3d("z",(0,0,scale+0.5),fontsize=30)
        plot30+=parametric_plot3d(param,(s,s1,s2),(t,t1,t2),color="gray",opacity=0.2)
        n=9;
        xvals=[-scale+i/n*2*scale for i in range(1,n)];
        #xvals=[0]
        for xtrace in xvals:
            if abs(xtrace)>a:
                tl=b*sqrt(scale^2+1-xtrace^2/a^2)
                xtparam(t)=(xtrace,t,sqrt(xtrace^2/a^2+t^2/b^2-1))
                xtparam2(t)=(xtrace,t,-sqrt(xtrace^2/a^2+t^2/b^2-1))
                plot30+=parametric_plot3d(xtparam(t),(t,-tl,tl),color="blue",thickness=1)                                                       
                plot30+=parametric_plot3d(xtparam2(t),(t,-tl,tl),color="blue",thickness=1)
            if abs(xtrace)<a:
                tl=sqrt(scale^2-1+xtrace^2/a^2)
                xtparam(t)=(xtrace,b*sqrt(1-xtrace^2/a^2+t^2),t)
                xtparam2(t)=(xtrace,-b*sqrt(1-xtrace^2/a^2+t^2),t)
                plot30+=parametric_plot3d(xtparam(t),(t,-tl,tl),color="blue",thickness=1)
                plot30+=parametric_plot3d(xtparam2(t),(t,-tl,tl),color="blue",thickness=1)
            if abs(xtrace)==a:
                xtparam(t)=(xtrace,b*t,t)
                xtparam2(t)=(xtrace,-b*t,t)
                tl=scale
                plot30+=parametric_plot3d(xtparam(t),(t,-tl,tl),color="blue",thickness=1)
                plot30+=parametric_plot3d(xtparam2(t),(t,-tl,tl),color="blue",thickness=1)
        yvals=[-scale*b+i/n*2*scale*b for i in range(1,n)];
        for ytrace in yvals:
            if abs(ytrace)>b:
                tl=a*sqrt(scale^2+1-ytrace^2/b^2)
                ytparam(t)=(t,ytrace,sqrt(ytrace^2/b^2+t^2/a^2-1))
                ytparam2(t)=(t,ytrace,-sqrt(ytrace^2/b^2+t^2/a^2-1))
                plot30+=parametric_plot3d(ytparam(t),(t,-tl,tl),color="green",thickness=1)
                plot30+=parametric_plot3d(ytparam2(t),(t,-tl,tl),color="green",thickness=1)
            if abs(ytrace)<b:
                tl=sqrt(scale^2-1+ytrace^2/b^2)
                ytparam(t)=(a*sqrt(1-ytrace^2/b^2+t^2),ytrace,t)
                ytparam2(t)=(-a*sqrt(1-ytrace^2/n^2+t^2),ytrace,t)
                plot30+=parametric_plot3d(ytparam(t),(t,-tl,tl),color="green",thickness=1)
                plot30+=parametric_plot3d(ytparam2(t),(t,-tl,tl),color="green",thickness=1)
            if abs(ytrace)==b:
                ytparam(t)=(a*t,ytrace,t)
                ytparam2(t)=(-a*t,ytrace,t)
                tl=scale
                plot30+=parametric_plot3d(ytparam(t),(t,-tl,tl),color="green",thickness=1)
                plot30+=parametric_plot3d(ytparam2(t),(t,-tl,tl),color="green",thickness=1)
        zvals=[-scale*c+i/n*2*scale*c for i in range(1,n)];
        for ztrace in zvals:
            plot30+=parametric_plot3d((a*sqrt(1+ztrace^2/c^2)*cos(t),b*sqrt(1+ztrace^2/c^2)*sin(t),ztrace),(t,0,2*pi),color="red") 
    if sb=='Hyperboloid of Two Sheets':
        scale=3
        a=1;b=1.25;c=1;
        param(s,t)=(a*s*cos(t),b*s*sin(t),c*sqrt(1+s^2));s1=0;s2=2.85;t1=0;t2=2*pi;
        param2(s,t)=(a*s*cos(t),b*s*sin(t),-c*sqrt(1+s^2));
        plot30=axes(scale,1.5,color="gray",frame=False,aspect_ratio=1)+text3d("x",(scale+0.5,0,0),fontsize=30)+text3d("y",(0,scale+0.5,0),fontsize=30)+text3d("z",(0,0,scale+0.5),fontsize=30)
        plot30+=parametric_plot3d(param,(s,s1,s2),(t,t1,t2),color="gray",opacity=0.2)
        plot30+=parametric_plot3d(param2,(s,s1,s2),(t,t1,t2),color="gray",opacity=0.2)
        n=11;
        xvals=[-a*scale+i/n*2*a*scale for i in range(1,n)];
        for xtrace in xvals:
            tl=b*sqrt(scale^2/c^2-xtrace^2/a^2-1)
            plot30+=parametric_plot3d((xtrace,t,sqrt(xtrace^2/a^2+t^2/b^2+1)),(t,-tl,tl),color="blue",thickness=1)
            plot30+=parametric_plot3d((xtrace,t,-sqrt(xtrace^2/a^2+t^2/b^2+1)),(t,-tl,tl),color="blue",thickness=1)
        yvals=[-b*scale+i/n*2*b*scale for i in range(1,n)];
        for ytrace in yvals:
            tl=a*sqrt(scale^2/c^2-ytrace^2/b^2-1)
            plot30+=parametric_plot3d((t,ytrace,sqrt(ytrace^2/b^2+t^2/a^2+1)),(t,-tl,tl),color="green",thickness=1)
            plot30+=parametric_plot3d((t,ytrace,-sqrt(ytrace^2/b^2+t^2/a^2+1)),(t,-tl,tl),color="green",thickness=1)
        zvals=[-scale*c+i/n*2*scale*c for i in range(0,n+1)];
        for ztrace in zvals:
            if abs(ztrace)>c:
                plot30+=parametric_plot3d((a*sqrt(-1+ztrace^2/c^2)*cos(t),b*sqrt(-1+ztrace^2/c^2)*sin(t),ztrace),(t,0,2*pi),color="red")      
    if sb=='Elliptic Paraboloid':
        scale=3
        a=1;b=1.5;c=1;
        param(s,t)=(a*s*cos(t),b*s*sin(t),c*s^2);s1=0;s2=sqrt(3);t1=0;t2=2*pi;
        plot30=axes(scale,1.5,color="gray",frame=False,aspect_ratio=1)+text3d("x",(scale+0.5,0,0),fontsize=30)+text3d("y",(0,scale+0.5,0),fontsize=30)+text3d("z",(0,0,scale+0.5),fontsize=30)
        plot30+=parametric_plot3d(param,(s,s1,s2),(t,t1,t2),color="gray",opacity=0.2)
        n=11;
        xvals=[-a*2+i/n*2*a*2 for i in range(1,n)];
        for xtrace in xvals:
            tl=b*sqrt(scale/c-xtrace^2/a^2)
            plot30+=parametric_plot3d((xtrace,t,xtrace^2/a^2+t^2/b^2),(t,-tl,tl),color="blue",thickness=1)
        yvals=[-b*2+i/n*2*b*2 for i in range(1,n)];
        for ytrace in yvals:
            tl=a*sqrt(scale/c-ytrace^2/b^2)
            plot30+=parametric_plot3d((t,ytrace,t^2/a^2+ytrace^2/b^2),(t,-tl,tl),color="green",thickness=1)
        n=6
        zvals=[i/n*scale*c for i in range(0,n+1)];
        for ztrace in zvals:
            plot30+=parametric_plot3d((a*sqrt(ztrace)*cos(t),b*sqrt(ztrace)*sin(t),ztrace),(t,0,2*pi),color="red")
    if sb=='Saddle/Hyperbolic Paraboloid':
        scale=3
        a=1;b=1.5;c=1;
        param(s,t)=(a*s*cos(t),b*s*sin(t),c*s^2*cos(2*t));s1=0;s2=sqrt(3);t1=0;t2=2*pi;
        plot30=axes(scale,1.5,color="gray",frame=False,aspect_ratio=1)+text3d("x",(scale+0.5,0,0),fontsize=30)+text3d("y",(0,scale+0.5,0),fontsize=30)+text3d("z",(0,0,scale+0.5),fontsize=30)
        plot30+=parametric_plot3d(param,(s,s1,s2),(t,t1,t2),color="gray",opacity=0.2,plot_points=101)

        n=10;
        xvals=[-a*2+i/n*2*a*2 for i in range(1,n)];
        for xtrace in xvals:
            tl=b*sqrt(-xtrace^2/a^2+scale)
            plot30+=parametric_plot3d((xtrace,t,xtrace^2/a^2-t^2/b^2),(t,-tl,tl),color="blue",thickness=1)
        yvals=[-b*2+i/n*2*b*2 for i in range(1,n)];
        for ytrace in yvals:
            tl=a*sqrt(-ytrace^2/b^2+scale)
            plot30+=parametric_plot3d((t,ytrace,t^2/a^2-ytrace^2/b^2),(t,-tl,tl),color="green",thickness=1)
        zvals=[-scale+i/n*2*scale for i in range(1,n)];
        #zvals=[-1]
        for ztrace in zvals:
            if sign(ztrace)==-1:
                tl=a*sqrt((3+ztrace)/2)
                plot30+=parametric_plot3d((t,b*sqrt(t^2/a^2-ztrace),ztrace),(t,-tl,tl),color="red",thickness=1)
                plot30+=parametric_plot3d((t,-b*sqrt(t^2/a^2-ztrace),ztrace),(t,-tl,tl),color="red",thickness=1)
            if sign(ztrace)==1:
                tl=b*sqrt((3-ztrace)/2)
                plot30+=parametric_plot3d((a*sqrt(t^2/b^2+ztrace),t,ztrace),(t,-tl,tl),color="red",thickness=1)
                plot30+=parametric_plot3d((-a*sqrt(t^2/b^2+ztrace),t,ztrace),(t,-tl,tl),color="red",thickness=1)
            if ztrace==0:
                plot30+=parametric_plot3d((a*t,b*t,0),(t,-sqrt(1.5),sqrt(1.5)),color="red")
                plot30+=parametric_plot3d((-a*t,b*t,0),(t,-sqrt(1.5),sqrt(1.5)),color="red")
  
    if sb=='Cone':
        scale=3
        a=1;b=1.5;c=1;
        param(s,t)=(a*s*cos(t),b*s*sin(t),c*s);s1=-3;s2=3;t1=0;t2=2*pi;
        plot30=axes(scale,1.5,color="gray",frame=False,aspect_ratio=1)+text3d("x",(scale+0.5,0,0),fontsize=30)+text3d("y",(0,scale+0.5,0),fontsize=30)+text3d("z",(0,0,scale+0.5),fontsize=30)
        plot30+=parametric_plot3d(param,(s,s1,s2),(t,t1,t2),color="gray",opacity=0.2)
        n=10;
        xvals=[-scale+i/n*2*scale for i in range(1,n)];
        for xtrace in xvals:
            if abs(xtrace)==0:
                xtparam(t)=(xtrace,b*t,t)
                xtparam2(t)=(xtrace,-b*t,t)
                tl=scale
                plot30+=parametric_plot3d(xtparam(t),(t,-tl,tl),color="blue",thickness=1)
                plot30+=parametric_plot3d(xtparam2(t),(t,-tl,tl),color="blue",thickness=1)
            else:
                tl=b*sqrt(scale^2-xtrace^2/a^2)
                xtparam(t)=(xtrace,t,sqrt(xtrace^2/a^2+t^2/b^2))
                xtparam2(t)=(xtrace,t,-sqrt(xtrace^2/a^2+t^2/b^2))
                plot30+=parametric_plot3d(xtparam(t),(t,-tl,tl),color="blue",thickness=1)
                plot30+=parametric_plot3d(xtparam2(t),(t,-tl,tl),color="blue",thickness=1)
        yvals=[-scale*b+i/n*2*scale*b for i in range(1,n)];
        for ytrace in yvals:
            if abs(ytrace)==0:
                ytparam(t)=(a*t,ytrace,t)
                ytparam2(t)=(-a*t,ytrace,t)
                tl=scale
                plot30+=parametric_plot3d(ytparam(t),(t,-tl,tl),color="green",thickness=1)
                plot30+=parametric_plot3d(ytparam2(t),(t,-tl,tl),color="green",thickness=1)
            else:
                tl=a*sqrt(scale^2-ytrace^2/b^2)
                xtparam(t)=(t,ytrace,sqrt(t^2/a^2+ytrace^2/b^2))
                xtparam2(t)=(t,ytrace,-sqrt(t^2/a^2+ytrace^2/b^2))
                plot30+=parametric_plot3d(xtparam(t),(t,-tl,tl),color="green",thickness=1)
                plot30+=parametric_plot3d(xtparam2(t),(t,-tl,tl),color="green",thickness=1)
        zvals=[-scale*c+i/n*2*scale*c for i in range(1,n)];
        for ztrace in zvals:
            plot30+=parametric_plot3d((a*ztrace*cos(t),b*ztrace*sin(t),ztrace),(t,0,2*pi),color="red")
    show(plot30,aspect_ratio=1)
