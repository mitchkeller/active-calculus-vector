\begin{align*} f_{xx} \amp= (f_x)_x = \frac{\partial}{\partial x} \Bigl[ \frac{\partial f}{\partial x} \Bigr] = \frac{\partial^2 f}{\partial x^2}\\ f_{yy} \amp= (f_y)_y=\frac{\partial}{\partial y} \Bigl[ \frac{\partial f}{\partial y} \Bigr] = \frac{\partial^2 f}{\partial y^2}\\ f_{xy} \amp= (f_x)_y=\frac{\partial}{\partial y} \Bigl[ \frac{\partial f}{\partial x} \Bigr] = \frac{\partial^2 f}{\partial y \partial x}\\ f_{yx}\amp=(f_y)_x=\frac{\partial}{\partial x} \Bigl[ \frac{\partial f}{\partial y} \Bigr] = \frac{\partial^2 f}{\partial x \partial y} \end{align*}
